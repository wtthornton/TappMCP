üí° Example Usage

Building a Web Application

// 1. Initialize project
const project = await smartBegin({
  projectName: "ECommerceApp",
  techStack: ["typescript", "react", "nodejs", "postgresql"],
  targetUsers: ["strategy-people", "vibe-coders"],
  businessGoals: ["increase sales", "improve user experience"],
  role: "developer",
  qualityLevel: "enterprise"
});

// 2. Create development plan
const plan = await smartPlan({
  projectId: project.projectId,
  planType: "development",
  scope: {
    features: ["product-catalog", "shopping-cart", "checkout"],
    timeline: { duration: 12 },
    resources: { teamSize: 6, budget: 100000 }
  },
  role: "product-strategist"
});

// 3. Generate code
const authCode = await smartWrite({
  projectId: project.projectId,
  featureDescription: "User authentication with JWT",
  targetRole: "developer",
  codeType: "api",
  techStack: ["typescript", "express", "jsonwebtoken"]
});

// 4. Validate quality
const validation = await smartFinish({
  projectId: project.projectId,
  codeIds: [authCode.codeId],
  qualityGates: {
    testCoverage: 90,
    securityScore: 95,
    complexityScore: 80
  },
  role: "qa-engineer"
});

// 5. Orchestrate deployment
const deployment = await smartOrchestrate({
  request: "Deploy ECommerceApp to production",
  options: {
    businessContext: {
      projectId: project.projectId,
      businessGoals: ["production-deployment", "zero-downtime"],
      requirements: ["docker", "aws", "ci-cd"]
    }
  },
  workflow: "sdlc",
  role: "operations-engineer"
});
